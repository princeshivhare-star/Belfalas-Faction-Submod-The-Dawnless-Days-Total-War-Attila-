Confederation — Lua Campaign Script

belfalas_confederation.lua

Complete Lua script — auto-loaded from script/campaign/mod/ on every campaign start.

script/campaign/mod/belfalas_confederation.lua
-- ================================================================
-- belfalas_confederation.lua  |  Belfalas Submod for TDD v1.0.1
-- Handles: start alliance, confederation trigger, narrative event
-- ================================================================

local BELFALAS   = "wh2_main_brt_belfalas"   -- update if TDD uses different key
local GONDOR     = "tdd_gondor"              -- verify against TDD factions_tables
local ATT_THRESH = 150                       -- minimum attitude to offer confederation
local CHK_TURNS  = 5                         -- check every N turns

-- ── helpers ──────────────────────────────────────────────────
local function fac(k) return cm:get_faction(k) end

local function alive()
  local b,g = fac(BELFALAS), fac(GONDOR)
  return b and g and not b:is_dead() and not g:is_dead()
end

-- ── confederation check ───────────────────────────────────────
local function check_confederation()
  if not alive() then return end
  local att = cm:get_faction_attitude_to_faction(BELFALAS, GONDOR)
  if att >= ATT_THRESH then
    cm:trigger_confederation_event(
      GONDOR, BELFALAS, "event_conf_gondor_belfalas"
    )
    out("[Belfalas] Confederation triggered — attitude=" .. att)
  end
end

-- ── turn-start periodic check ─────────────────────────────────
cm:add_turn_start_callback(function()
  if cm:turn_number() % CHK_TURNS == 0 then
    check_confederation()
  end
end, true)

-- ── diplomacy change listener ─────────────────────────────────
core:add_listener(
  "belfalas_diplo_watch",
  "DiplomacyTurnEnd",
  function(ctx)
    local n = ctx:faction():name()
    return n == BELFALAS or n == GONDOR
  end,
  function() check_confederation() end,
  true
)

-- ── campaign start: force alliance + narrative event ──────────
core:add_listener(
  "belfalas_campaign_start",
  "ScriptEventAllFactions",
  true,
  function()
    if not alive() then return end
    -- guarantee military alliance from turn 1
    if not cm:factions_are_allies(BELFALAS, GONDOR) then
      cm:force_diplomacy(
        BELFALAS, GONDOR, "military_alliance", true, true
      )
      out("[Belfalas] Start alliance with Gondor — applied.")
    end
  end,
  false   -- fire once only
)

-- ── turn-1 opening narrative ──────────────────────────────────
core:add_listener(
  "belfalas_turn1_event",
  "FactionTurnStart",
  function(ctx)
    return ctx:faction():name() == BELFALAS
      and cm:turn_number() == 1
  end,
  function()
    cm:show_message_event(
      BELFALAS,
      "event_title_belfalas_dawn",
      "event_body_belfalas_dawn",
      "event_advice_belfalas_dawn",
      true
    )
  end,
  false
)

out("[Belfalas] belfalas_confederation.lua — loaded OK.")
